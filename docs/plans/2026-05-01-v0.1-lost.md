## Rejected alternatives

- **Per-handler event rolls** — each terrain handler (`onEnterWoods` / `onEnterMountain` / `onEnterSwamp`) rolls its own events. Rejected: today woods/mountain share `onEnterDefaultTerrain`; per-terrain handlers would be a wider refactor with no clear gain. The single `rollTileEvent` function is enough.
- **Lost as a tile-enter handler outcome (`outcome.teleport: Vec2`)** — handlers return their own teleport target. Rejected: teleport is a position-changing event sibling of combat (which already lives in `reduceMove` via the encounter pipeline), not a local tile effect. Keeping `lost` next to `fight` in the event union is more cohesive.
- **Two independent rolls (combat AND lost)** — separate ambush/lost rolls run in sequence. Rejected by user during brainstorm: stacks total event density (~50% on woods if 25% + 25%). Single roll is structurally mutex.
```

  // move enters `center`. Grass is non-orienting, non-lost, non-fight, so the

  return {

      [grass(), grass(), grass(), grass(), grass()],
      [grass(), grass(), opts.center, grass(), grass()],
      [grass(), grass(), grass(), grass(), grass()],
      [grass(), grass(), grass(), grass(), grass()],
    ],
    rngState: opts.rngState ?? 12345,
  }
}

function makeState(world: World): State {
  return {
    world,
    player: { position: { x: 2, y: 1 } },
    run: { stepCount: 0, hasWon: false, isGameOver: false, knowsPosition: false },
    resources: { food: INITIAL_FOOD, armySize: 5, hasBronzeKey: false },
    encounter: null,
    ui: { message: '', leftPanel: { kind: 'auto' }, clock: { frame: 0 }, anim: { nextId: 1, active: [] } },
  }
}

// Find a {seed, stepCount, cellId} where the percentile lands in [lo, hi).
function findEventSeed(opts: { stepCount: number; cellId: number; lo: number; hi: number }): number {
  for (let seed = 1; seed < 200000; seed++) {
    const h = hashSeedStepCell({ seed, stepCount: opts.stepCount, cellId: opts.cellId })
    const p = h % 100
    if (p >= opts.lo && p < opts.hi) return seed
  }
  throw new Error('no seed found in range')
}

import {
  MOUNTAIN_AMBUSH_PERCENT,
  SWAMP_LOST_PERCENT,
  WOODS_AMBUSH_PERCENT,
  WOODS_LOST_PERCENT,
} from './constants'
import { hashSeedStepCell } from './prng'
import type { CellKind } from './types'

export type TileEventSource = 'woods' | 'mountain' | 'swamp' | 'henge'

export type TileEvent =
  | { kind: 'fight'; source: TileEventSource }
  | { kind: 'lost'; source: 'woods' | 'swamp' }

export function rollTileEvent(args: {
  seed: number
  stepCount: number
  cellId: number
  kind: CellKind
  hengeReady: boolean
}): TileEvent | null {
  const { seed, stepCount, cellId, kind, hengeReady } = args

  if (kind === 'henge') {
    return hengeReady ? { kind: 'fight', source: 'henge' } : null
  }

  let ambushPct = 0
  let lostPct = 0
  if (kind === 'woods') { ambushPct = WOODS_AMBUSH_PERCENT; lostPct = WOODS_LOST_PERCENT }
  else if (kind === 'mountain') { ambushPct = MOUNTAIN_AMBUSH_PERCENT; lostPct = 0 }
  else if (kind === 'swamp') { ambushPct = 0; lostPct = SWAMP_LOST_PERCENT }
  else return null

  if (ambushPct + lostPct === 0) return null
  const p = hashSeedStepCell({ seed, stepCount, cellId }) % 100
  if (p < ambushPct) return { kind: 'fight', source: kind as TileEventSource }
  if (p < ambushPct + lostPct) return { kind: 'lost', source: kind as 'woods' | 'swamp' }
  return null
}
```

```ts
import { TELEPORT_MIN_DISTANCE } from './constants'
import { manhattan, torusDelta } from './math'
import { randInt } from './prng'
import type { Vec2, World } from './types'

const NON_TERRAIN_KINDS = new Set(['gate', 'gateOpen', 'locksmith', 'signpost', 'farm', 'camp', 'henge'])

function isTerrain(kind: string): boolean {
  return !NON_TERRAIN_KINDS.has(kind)
}

export function pickTeleportDestination(args: {
  world: World
  origin: Vec2
  rngState: number
}): { destination: Vec2; rngState: number } {
  const { world, origin } = args
  let rngState = args.rngState

  type Candidate = { x: number; y: number; d: number }
  const candidates: Candidate[] = []
  let maxD = 0

  for (let y = 0; y < world.height; y++) {
    const row = world.cells[y]!
    for (let x = 0; x < world.width; x++) {
      const cell = row[x]!
      if (!isTerrain(cell.kind)) continue
      if (x === origin.x && y === origin.y) continue
      const dx = torusDelta(origin.x, x, world.width)
      const dy = torusDelta(origin.y, y, world.height)
      const d = manhattan(dx, dy)
      candidates.push({ x, y, d })
      if (d > maxD) maxD = d
    }
  }

  const target = Math.min(TELEPORT_MIN_DISTANCE, maxD)
  const eligible = candidates.filter((c) => c.d >= target)
  // If somehow eligible is empty (no non-feature terrain), fall back to all candidates.
  const pool = eligible.length > 0 ? eligible : candidates
  if (pool.length === 0) {
    // Degenerate: no non-feature terrain at all. Return origin to keep state sane.
    return { destination: origin, rngState }
  }

  const r = randInt(rngState, pool.length)
  rngState = r.rngState
  const pick = pool[r.value]!
  return { destination: { x: pick.x, y: pick.y }, rngState }
}
```
