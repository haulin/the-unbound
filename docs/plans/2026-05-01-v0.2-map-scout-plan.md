## As-implemented notes (post-plan refactors)

- Map rendering is now a **rolling, player-centered viewport** (torus-wrapped) rather than a whole-world grid. The player marker is always centered and does not reveal global position while lost. See `src/platform/tic80/render.ts`.
- Map tile background uses sprite `#135` (colorkey 0) stamped per cell; player outline uses sprite `#134` (colorkey 0).
- Map button sprite is `#138` (top-right corner). Camp encounter center icon uses `#140`.
- Lore/name pools (plus barkeep tips and the mechanics index) are centralized in `src/core/lore.ts`. `src/core/constants.ts` re-exports them.
- Map toggle temporarily overrides `ui.message` (`MAP_HINT_MESSAGE`); closing restores the previous message if unchanged.

**Reasoning:** Implement the approved v0.2 design using existing deterministic helpers:

- `hashSeedStepCell` (`src/core/prng.ts`)
- `pickDeterministicLine` (`src/core/tiles/poiUtils.ts`)

Camp offer + Search must be deterministic and must **not** consume `world.rngState`. Do **not** duplicate existing camp lore pools (`CAMP_RECRUIT_LINES`, `CAMP_EMPTY_LINES`). If any new PRNG helper is required, it lives in `src/core/prng.ts`.

---

**Goal:** Add an in‑game landmarks map, a Camp encounter with Search/Hire/Leave actions, Scout acquisition + HUD icon, Scout map reveal + reduced lost chance, and replay‑ready path mapping with buffered/backfill/discard semantics while lost.

**Architecture:**

- Pure helpers:
  - `src/core/camp.ts` — deterministic camp offer (`armyGain`) via `hashSeedStepCell`
  - `src/core/gameMap.ts` — compute map markers + whether to show player marker
- State changes:
  - `resources.hasScout`
  - `run.path: Array<{ pos: Vec2; isMapped: boolean }>`
  - `run.lostBufferStartIndex: number | null`
  - `Encounter` adds `camp`
  - `LeftPanel` adds `map`
- Reducer integrates:
  - camp encounter entry + actions (camp actions do **not** increment `run.stepCount`)
  - map toggle (available everywhere; restores previous left-panel mode on close)
  - run.path mapping rules
  - passes `hasScout` to tile-event roll to halve lost chance

**Tech Stack:** TypeScript + Vitest; TIC‑80 renderer under `src/platform/tic80/`.

**TDD during implementation:** waive — evolving hobby project; tests still required, but strict RED→GREEN ordering is not mandatory for every change

**ATDD during implementation:** waive — evolving hobby project; acceptance tests are used selectively for stable behaviors, unit tests are the primary safety net

## Testing approach (tests required; unit-first; selective acceptance)

We will still write tests for the work, but we will not treat the full GWT list as a strict “acceptance suite must exist for every behavior” contract in this phase.

- **Unit tests are primary** for fast iteration (`tests/core/*.test.ts`).
- **Acceptance-style tests are selective** (`tests/core/v0.2-map-scout.acceptance.test.ts`) for the most stable and regression-prone behaviors (camp encounter, hire/search flows, path semantics, map toggles).

Each behavior task should:

- Add/extend an acceptance test (RED).
- Add/extend a focused unit test for pure helpers (where applicable) (RED).
- Implement minimal code to pass.
- Run the focused tests, then the full suite (`npm test`) before moving on.

Commands used throughout:

- `npm test -- <path>` for focused tests
- `npm run verify` for the final gate (typecheck + build + full suite)

```ts
import { hashSeedStepCell } from './prng'

export function computeCampArmyGain(args: { seed: number; campId: number; stepCount: number }): number {
  const h = hashSeedStepCell({ seed: args.seed, stepCount: args.stepCount, cellId: args.campId })
  return 1 + (h % 2)
}
```

import { processAction } from '../../src/core/processAction'
import {
  ACTION_CAMP_HIRE_SCOUT,
  ACTION_CAMP_LEAVE,
  ACTION_CAMP_SEARCH,
  ACTION_MOVE,
  ACTION_TOGGLE_MAP,
  ACTION_TOGGLE_MINIMAP,

  CAMP_RECRUIT_LINES,
  INITIAL_FOOD,

import type { Cell, State, World } from '../../src/core/types'

function grass(): Cell {
  return { kind: 'grass' }
}

function makeWorld(seed: number, campNextReadyStep = 0): World {
  return {
    seed,
    width: 3,
    height: 3,
    mapGenAlgorithm: 'TEST',
    cells: [
      [grass(), grass(), grass()],
      [grass(), { kind: 'camp', id: 4, name: 'Ember Cross', nextReadyStep: campNextReadyStep }, grass()],
      [grass(), grass(), grass()],
    ],
    rngState: 123,
  }
}

function makeState(world: World): State {
  return {
    world,
    player: { position: { x: 1, y: 0 } },
    run: { stepCount: 0, hasWon: false, isGameOver: false, knowsPosition: false, path: [], lostBufferStartIndex: null },

describe('v0.2 map+scout acceptance', () => {
  it('stepping onto a camp enters a camp encounter; MOVE ignored until Leave', () => {
    const s0 = makeState(makeWorld(7))
    const onto = processAction(s0, { type: ACTION_MOVE, dx: 0, dy: 1 })!
    expect(onto.encounter?.kind).toBe('camp')

    const ignored = processAction(onto, { type: ACTION_MOVE, dx: 1, dy: 0 })!
    expect(ignored).toBe(onto)

    const left = processAction(onto, { type: ACTION_CAMP_LEAVE })!
    expect(left.encounter).toBe(null)

    expect(onto.encounter).toBe(null)

    const s0 = makeState(makeWorld(7))
    const onto = processAction(s0, { type: ACTION_MOVE, dx: 0, dy: 1 })!
    const next = processAction(onto, { type: ACTION_TOGGLE_MAP })!
    expect(next).toBe(onto)
  })

  it('map/minimap toggles are mutually exclusive (last toggle wins)', () => {
    const s0 = makeState(makeWorld(7))
    const mini = processAction(s0, { type: ACTION_TOGGLE_MINIMAP })!
    expect(mini.ui.leftPanel.kind).toBe('minimap')

    const map = processAction(mini, { type: ACTION_TOGGLE_MAP })!
    expect(map.ui.leftPanel.kind).toBe('map')

    const mini2 = processAction(map, { type: ACTION_TOGGLE_MINIMAP })!
    expect(mini2.ui.leftPanel.kind).toBe('minimap')
  })

  it('Search camp grants deterministic reinforcements when ready; then becomes empty for the second Search', () => {
    const s0 = makeState(makeWorld(7, 0))
    const onto = processAction(s0, { type: ACTION_MOVE, dx: 0, dy: 1 })!
    const stepCount = onto.run.stepCount
    const armyGain = computeCampArmyGain({ seed: onto.world.seed, campId: 4, stepCount })

    const after = processAction(onto, { type: ACTION_CAMP_SEARCH })!
    expect(after.resources.food).toBe(onto.resources.food + CAMP_FOOD_GAIN)
    expect(after.resources.armySize).toBe(onto.resources.armySize + armyGain)
    expect(after.ui.message).toBe(

    onto.resources.food = 999

    const after = processAction(onto, { type: ACTION_CAMP_HIRE_SCOUT })!
    expect(after.resources).toEqual(onto.resources)
    expect(after.ui.message).toBe(
      `Ember Cross Camp\n${pickDeterministicLine(SCOUT_ALREADY_HAVE_LINES, onto.world.seed, 4, stepCount)}`,
    )
  })

  it('Hire Scout without enough food shows lore and changes nothing', () => {
    const s0 = makeState(makeWorld(7))
    const onto = processAction(s0, { type: ACTION_MOVE, dx: 0, dy: 1 })!
    const stepCount = onto.run.stepCount

    expect(after.resources.food).toBe(0)
    expect(after.ui.message).toBe(`Ember Cross Camp\n${pickDeterministicLine(SCOUT_NO_FOOD_LINES, onto.world.seed, 4, stepCount)}`)
  })
})

```ts
import type { CellKind, State, Vec2 } from './types'

export type GameMapLabel = 'F' | 'C' | 'H' | 'G' | 'L'
export type GameMapMarker = { pos: Vec2; label: GameMapLabel }

function labelForKind(kind: CellKind): GameMapLabel | null {
  if (kind === 'farm') return 'F'
  if (kind === 'camp') return 'C'
  if (kind === 'henge') return 'H'
  if (kind === 'gate') return 'G'
  if (kind === 'locksmith') return 'L'
  return null
}

export function computeGameMapView(s: State): { markers: GameMapMarker[]; showPlayer: boolean } {
  const showPlayer = !!s.run.knowsPosition
  const markers: GameMapMarker[] = []
  const seen = new Set<string>()

  function push(pos: Vec2, label: GameMapLabel) {
    const k = `${label}@${pos.x},${pos.y}`
    if (seen.has(k)) return
    seen.add(k)
    markers.push({ pos, label })
  }

  // With scout: global reveal for F/C/H.
  if (s.resources.hasScout) {
    for (let y = 0; y < s.world.height; y++) {
      for (let x = 0; x < s.world.width; x++) {
        const kind = s.world.cells[y]![x]!.kind
        const label = kind === 'farm' ? 'F' : kind === 'camp' ? 'C' : kind === 'henge' ? 'H' : null
        if (label) push({ x, y }, label)
      }
    }
  }

  // Baseline: mapped run.path positions.
  for (let i = 0; i < s.run.path.length; i++) {
    const step = s.run.path[i]!
    if (!step.isMapped) continue
    const p = step.pos
    const kind = s.world.cells[p.y]![p.x]!.kind
    const label = labelForKind(kind)
    if (!label) continue

    // With scout: keep G/L gated by mappedness, but allow F/C/H to show as well.
    if (s.resources.hasScout && (label === 'G' || label === 'L')) push(p, label)
    else if (!s.resources.hasScout) push(p, label)
    else if (label === 'F' || label === 'C' || label === 'H') push(p, label)
  }

  return { markers, showPlayer }
}
```
