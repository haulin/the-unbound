# v0.3 ‚Äî Gold & Towns Implementation Plan

Canonical implementation plan for v0.3 (kept for historical record).

Design reference: `docs/plans/2026-05-02-v0.3-gold-towns-design.md`

## Sprite IDs (v0.3)

Sprite IDs are centralized in `src/core/spriteIds.ts` as `SPRITES` (single source of truth). Game code references `SPRITES.*` directly.

- **Town right-grid sprites**:
  - Food: `SPRITES.buttons.food` (`130`)
  - Troops: `SPRITES.buttons.troop` (`104`)
  - Scout: `SPRITES.buttons.scout` (`134`)
  - Rumor: `SPRITES.buttons.rumorTip` (`136`)
  - Center: `SPRITES.cosmetics.marketStall` (`200`)
- **Town preview plate rumor art**:
  - `SPRITES.cosmetics.rumorIllustration` (`168`)

Note: the detailed task list below is kept for historical record and includes older constant names and values (e.g. `TILE_*`, `*_SPRITE_ID`, `RIGHT_GRID_SPRITE_ID`, `iconKey`, initial balance ranges). For the current truth, refer to `src/core/constants.ts` and `src/core/spriteIds.ts`.

## Post-implementation tweaks (record)

- **Map**: towns show as `T`; `gateOpen` still shows as `G` (`src/core/gameMap.ts`).
- **Signposts**: towns are eligible PoIs and ranked between farms and camps (`src/core/signpost.ts`).
- **Rumors**: rumor pool is derived from `BARKEEP_TIPS` in `town.ts`, shuffled per town, and cycles on repeated purchase (`src/core/town.ts`).
- **Town preview**: offer price plate shows costs with stat icons, rumor illustration sprite is `SPRITES.cosmetics.rumorIllustration` (`src/platform/tic80/render.ts`, `src/core/spriteIds.ts`).
- **Top bar**: right-panel header spans grid width and is drawn after grid ops so it doesn't disappear during tile slide (`src/platform/tic80/render.ts`).
- **Sprite ID refactor**: sprite numbers moved into `src/core/spriteIds.ts` and right-grid rendering simplified to use `spriteId` directly (no `iconKey` indirection).
- **Playtest tuning**: increased starting gold to `INITIAL_GOLD=15`, increased combat gold reward to `8..20` and food bonus to `0..4`, and reduced town scout price range to `8..12` (see `src/core/constants.ts`).

## Execution mode

Use **`/use-subagents`**: one small task per subagent, run tests per task, then a quick review pass.

import { describe, expect, it } from 'vitest'
import { generateWorld } from '../../src/core/world'
import { TOWN_COUNT } from '../../src/core/constants'

function flatten(world: { width: number; height: number; cells: any[][] }) {
  const out: any[] = []
  for (let y = 0; y < world.height; y++) {
    for (let x = 0; x < world.width; x++) out.push(world.cells[y]![x]!)
  }
  return out
}

describe('world towns', () => {
  it('places towns deterministically and guarantees hireScout exists', () => {
    const w1 = generateWorld(123).world
    const w2 = generateWorld(123).world

    const towns1 = flatten(w1).filter((c) => c.kind === 'town')
    const towns2 = flatten(w2).filter((c) => c.kind === 'town')

    expect(towns1.length).toBe(TOWN_COUNT)
    expect(towns2.length).toBe(TOWN_COUNT)

    // Stable structured fields.
    expect(towns1.map((t: any) => ({ id: t.id, offers: t.offers, prices: t.prices }))).toEqual(
      towns2.map((t: any) => ({ id: t.id, offers: t.offers, prices: t.prices })),
    )

    // Scout guarantee.
    expect(towns1.some((t: any) => Array.isArray(t.offers) && t.offers.includes('hireScout'))).toBe(true)
  })

```ts
import {
  ACTION_TOWN_BUY_FOOD,
  ACTION_TOWN_BUY_RUMOR,
  ACTION_TOWN_BUY_TROOPS,
  ACTION_TOWN_HIRE_SCOUT,
  ACTION_TOWN_LEAVE,
  ENABLE_ANIMATIONS,
  FOOD_DELTA_FRAMES,
  GRID_TRANSITION_STEP_FRAMES,
  TOWN_BUY_LINES,
  TOWN_NO_GOLD_LINES,
  TOWN_RUMOR_LINES,
  TOWN_SCOUT_ALREADY_HAVE_LINES,
  TOWN_SCOUT_HIRE_LINES,
} from './constants'
import type { Action, Resources, State, TownCell, Ui } from './types'
import { pickDeterministicLine } from './tiles/poiUtils'
import { enqueueAnim } from './uiAnim'

function gridTransitionDurationFrames(): number {
  return Math.max(1, Math.trunc(GRID_TRANSITION_STEP_FRAMES)) * 5
}

function townAtPlayer(s: State): TownCell | null {
  const p = s.player.position
  const cell = s.world.cells[p.y]![p.x]!
  return cell.kind === 'town' ? (cell as TownCell) : null
}

function townPrefix(town: TownCell): string {
  const name = town.name || 'A Town'
  return `${name} Town`
}

export function reduceTownAction(prevState: State, action: Action): State | null {
  if (
    action.type !== ACTION_TOWN_BUY_FOOD &&
    action.type !== ACTION_TOWN_BUY_TROOPS &&
    action.type !== ACTION_TOWN_HIRE_SCOUT &&
    action.type !== ACTION_TOWN_BUY_RUMOR &&
    action.type !== ACTION_TOWN_LEAVE
  ) {
    return null
  }

  const enc = prevState.encounter
  if (!enc || enc.kind !== 'town') return prevState

  const town = townAtPlayer(prevState)
  if (!town) return prevState

  const stepCount = prevState.run.stepCount
  const seed = prevState.world.seed
  const townId = town.id
  const prefix = townPrefix(town)

  const prevRes = prevState.resources

  const setMessage = (line: string) => ({ ...prevState, ui: { ...prevState.ui, message: `${prefix}\n${line}` } })

  const noGold = () => setMessage(pickDeterministicLine(TOWN_NO_GOLD_LINES, seed, townId, stepCount))
  const genericBuy = () => pickDeterministicLine(TOWN_BUY_LINES, seed, townId, stepCount)

  if (action.type === ACTION_TOWN_LEAVE) {
    const restore = enc.restoreMessage
    const baseUi: Ui = { ...prevState.ui, message: restore }
    if (!ENABLE_ANIMATIONS) return { ...prevState, encounter: null, ui: baseUi }
    const startFrame = baseUi.clock.frame
    const uiWith = enqueueAnim(baseUi, {
      kind: 'gridTransition',
      startFrame,
      durationFrames: gridTransitionDurationFrames(),
      blocksInput: true,
      params: { from: 'town', to: 'overworld' },
    })
    return { ...prevState, encounter: null, ui: uiWith }
  }

  if (action.type === ACTION_TOWN_BUY_FOOD) {
    const cost = town.prices.foodGold
    if (prevRes.gold < cost) return noGold()
    const nextResources: Resources = { ...prevRes, gold: prevRes.gold - cost, food: prevRes.food + town.bundles.food }
    const baseUi: Ui = { ...prevState.ui, message: `${prefix}\n${genericBuy()}` }
    if (!ENABLE_ANIMATIONS) return { ...prevState, resources: nextResources, ui: baseUi }
    const startFrame = baseUi.clock.frame
    let uiWith = baseUi
    uiWith = enqueueAnim(uiWith, { kind: 'goldDelta', startFrame, durationFrames: FOOD_DELTA_FRAMES, blocksInput: false, params: { delta: -cost } })
    uiWith = enqueueAnim(uiWith, {
      kind: 'foodDelta',
      startFrame,
      durationFrames: FOOD_DELTA_FRAMES,
      blocksInput: false,
      params: { delta: town.bundles.food },
    })
    return { ...prevState, resources: nextResources, ui: uiWith }
  }

  if (action.type === ACTION_TOWN_BUY_TROOPS) {
    const cost = town.prices.troopsGold
    if (prevRes.gold < cost) return noGold()
    const nextResources: Resources = { ...prevRes, gold: prevRes.gold - cost, armySize: prevRes.armySize + town.bundles.troops }
    const baseUi: Ui = { ...prevState.ui, message: `${prefix}\n${genericBuy()}` }
    if (!ENABLE_ANIMATIONS) return { ...prevState, resources: nextResources, ui: baseUi }
    const startFrame = baseUi.clock.frame
    let uiWith = baseUi
    uiWith = enqueueAnim(uiWith, { kind: 'goldDelta', startFrame, durationFrames: FOOD_DELTA_FRAMES, blocksInput: false, params: { delta: -cost } })
    uiWith = enqueueAnim(uiWith, {
      kind: 'armyDelta',
      startFrame,
      durationFrames: FOOD_DELTA_FRAMES,
      blocksInput: false,
      params: { delta: town.bundles.troops },
    })
    return { ...prevState, resources: nextResources, ui: uiWith }
  }

  if (action.type === ACTION_TOWN_HIRE_SCOUT) {
    if (prevRes.hasScout) {
      return setMessage(pickDeterministicLine(TOWN_SCOUT_ALREADY_HAVE_LINES, seed, townId, stepCount))
    }
    const cost = town.prices.scoutGold
    if (prevRes.gold < cost) return noGold()
    const nextResources: Resources = { ...prevRes, hasScout: true, gold: prevRes.gold - cost }
    const line = pickDeterministicLine(TOWN_SCOUT_HIRE_LINES, seed, townId, stepCount)
    const baseUi: Ui = { ...prevState.ui, message: `${prefix}\n${line}` }
    if (!ENABLE_ANIMATIONS) return { ...prevState, resources: nextResources, ui: baseUi }
    const startFrame = baseUi.clock.frame
    const uiWith = enqueueAnim(baseUi, { kind: 'goldDelta', startFrame, durationFrames: FOOD_DELTA_FRAMES, blocksInput: false, params: { delta: -cost } })
    return { ...prevState, resources: nextResources, ui: uiWith }
  }

  // ACTION_TOWN_BUY_RUMOR
  const cost = town.prices.rumorGold
  if (prevRes.gold < cost) return noGold()
  const rumor = pickDeterministicLine(TOWN_RUMOR_LINES, seed, townId, stepCount)
  const nextResources: Resources = { ...prevRes, gold: prevRes.gold - cost }
  const baseUi: Ui = { ...prevState.ui, message: `${prefix}\n${rumor}` }
  if (!ENABLE_ANIMATIONS) return { ...prevState, resources: nextResources, ui: baseUi }
  const startFrame = baseUi.clock.frame
  const uiWith = enqueueAnim(baseUi, { kind: 'goldDelta', startFrame, durationFrames: FOOD_DELTA_FRAMES, blocksInput: false, params: { delta: -cost } })
  return { ...prevState, resources: nextResources, ui: uiWith }
}
```
