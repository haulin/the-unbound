Lakes/rainbow ends become *placed* features to avoid balance explosion. Farms + Locksmith become modal encounters implemented via `state.encounter` (MOVE blocked until Leave).

Modal interactions use new encounter kinds with explicit actions + right-grid mapping; immediate-payoff features use `onEnter` handlers and mutate the destination cell to store cooldown/spent state.

```ts
import type { MechanicDef, TileEnterHandler } from '../types'
import { RNG } from '../../rng'
import { getCellAt, setCellAt } from '../../cells'
import { FISHING_LAKE_COOLDOWN_MOVES, FISHING_LAKE_READY_LINES, FISHING_LAKE_COOLDOWN_LINES } from '../../constants'

const onEnterFishingLake: TileEnterHandler = ({ cell, world, pos, stepCount, resources }) => {
  if (cell.kind !== 'fishingLake') return { message: '' }

  const lake = getCellAt(world, pos)
  if (!lake || lake.kind !== 'fishingLake') return { message: '' }

  const readyAt = lake.nextReadyStep ?? 0
  const r = RNG.createTileRandom({ world, stepCount, pos })

  if (stepCount < readyAt) {
    return { message: r.perMoveLine(FISHING_LAKE_COOLDOWN_LINES, { cellId: lake.id }) }
  }

  const sr = RNG.createStreamRandom(world.rngState)
  const gain = sr.intExclusive(3) + 1

  const nextWorld = setCellAt(
    { ...world, rngState: sr.rngState },
    pos,
    { ...lake, nextReadyStep: stepCount + FISHING_LAKE_COOLDOWN_MOVES },
  )

  return {
    world: nextWorld,
    resources: { ...resources, food: resources.food + gain },
    foodDeltas: [gain],
    message: r.perMoveLine(FISHING_LAKE_READY_LINES, { cellId: lake.id }),
  }
}

export const fishingLakeMechanic: MechanicDef = {
  id: 'fishingLake',
  kinds: ['fishingLake'],
  onEnter: onEnterFishingLake,
}
```

import type { MechanicDef, TileEnterHandler } from '../types'
import { RNG } from '../../rng'
import { getCellAt, setCellAt } from '../../cells'
import { RAINBOW_END_GOLD_PAYOUT, RAINBOW_END_PAYOUT_LINES, RAINBOW_END_SPENT_LINES } from '../../constants'

const onEnterRainbowEnd: TileEnterHandler = ({ cell, world, pos, stepCount, resources }) => {
  if (cell.kind !== 'rainbowEnd') return { message: '' }

  const re = getCellAt(world, pos)
  if (!re || re.kind !== 'rainbowEnd') return { message: '' }

  const r = RNG.createTileRandom({ world, stepCount, pos })
  if (re.hasPaidOut) {
    return { message: r.perMoveLine(RAINBOW_END_SPENT_LINES, { cellId: re.id }) }
  }

  const nextWorld = setCellAt(world, pos, { ...re, hasPaidOut: true })
  return {
    world: nextWorld,
    resources: { ...resources, gold: resources.gold + RAINBOW_END_GOLD_PAYOUT },
    goldDeltas: [RAINBOW_END_GOLD_PAYOUT],
    message: r.perMoveLine(RAINBOW_END_PAYOUT_LINES, { cellId: re.id }),
  }
}

export const rainbowEndMechanic: MechanicDef = {
  id: 'rainbowEnd',
  kinds: ['rainbowEnd'],
  mapLabel: 'R',
  onEnter: onEnterRainbowEnd,
}

import { BEAST_CARRY_CAP_BONUS } from './constants'

export function foodCarryCap(res: { armySize: number; hasTameBeast: boolean }): number {
  const base = Math.max(0, (res.armySize | 0) * 2)
  return base + (res.hasTameBeast ? BEAST_CARRY_CAP_BONUS : 0)
}

export function canAddFood(res: { food: number; armySize: number; hasTameBeast: boolean }, amount: number): boolean {
  const a = Math.max(0, amount | 0)
  return (res.food | 0) + a <= foodCarryCap(res)
}
